    "sizes": ["S","M","L"], "colors": ["trắng","đen"], "imageUrl": "https://..." }

  { "productId": 1, "name": "Áo thun", "price": 199000, "quantity": 2 }

POST /orders (COD):
  { "items": [{"productId":1,"name":"Áo thun","price":199000,"quantity":2}],
    "total": 398000,
    "shippingName": "Nguyễn Văn A",
    "shippingPhone": "0901234567",
    "shippingAddress": "123 Đường ABC, TP.HCM",
    "paymentMethod": "cod" }

POST /orders (VNPay):
  { ...same as above, "paymentMethod": "vnpay" }
  → Response: { "order": {...}, "paymentUrl": "https://sandbox.vnpayment.vn/..." }
  → Frontend redirect user đến paymentUrl
